üîπ Advanced Python (OOP & Flask Web Apps)

Learn Object-Oriented Programming (OOP) in Python.

class Car:
    def __init__(self, brand, model):
        self.brand = brand
        self.model = model

    def details(self):
        return f"{self.brand} {self.model}"

my_car = Car("Toyota", "Supra")
print(my_car.details())
